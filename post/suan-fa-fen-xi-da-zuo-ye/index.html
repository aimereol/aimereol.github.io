第一题：计算数组的反转

大致思路：

  1. 首先是生成随机数
    void sj(int n, int *x)
    {
    	int t, k;
    	int i;
    	srand((unsigned)time(NULL));
    	for (i = 0; i < 200000; i++)
    	{
    		x[i].data = i;
    	}
    	for (i = 0; i < n; i++)
    	{
    		t = rand() % (200000 - i);
    		k = x[i].data;
    		x[i].data = x[t].data;
    		x[t].data = k;
    	}
    	cout << "随机数产生了" << endl;
    }//生成随机数
  1. 利用分治法：
    void sf(int *x,int left,int right,int &fz)
    {
        int len=right-left;
    	if(len>1) 
    	{
    		sf(x,left,(left + right)/2,fz);
    		sf(x,((left + right)/2)+1,right,fz);
    	}
    	sort(x, left, right, fz, len);
    }//当分组长度小于等于2时开始合并排序，并计算反转
